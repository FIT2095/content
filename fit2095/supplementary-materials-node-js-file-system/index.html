Question: is it possible to rename a file, create a directory, or even copy set of files in Node.js Answer: YES!!! using the fs built-in module
  The Node.js file system module allows you to work with the file system on your computer.   But how to add the fs module to my application?? 
let fs = require('fs');
 

Read Files Asynchronously

let fs = require('fs')

// this is the file name
let fileName = './data.txt';

//readFile is the function that reads yoru file
// utf8 because the file is text
fs.readFile(fileName, 'utf8', function (error, content) { 
    // if an error occurs, parameter error will get a value 
    if (error)
        console.log('We got an error!!!' + error);
    else {
        console.log('the content is:');
        console.log(content);
    }

})
 

Read Files Synchronously

let content = fs.readFileSync(fileName).toString();
console.log(content);
In this fragment, content will be set to the contents of the file, and no JavaScript code will be executed while the file is being read.  

Create a Directory (Folder)

fs.mkdir('testDirName', function (error) {
    if (error)
        console.log('An error has just occurred' + error);
    else {
        console.log('Directory created successfully!');
    }
});
 

Delete a Directory

fs.rmdir('testDirName', function (error) {
    if (error)
        console.log('An error has just occurred  ' + error);
    else
        console.log('Directory Deleted successfully!');
})
 

Create a file

fs.open('testFile.txt', 'w', function (error) {
    if (error)
        console.log('An error has just occurred  ' + error);
})
where
  • 'testFile.txt' is the name of the file to be created
  • 'w': open for write
  • function: callback to be invoked after the open operation
 

Rename Files

fs.rename('testWrite.txt', 'myrenamedfile.txt', function (err) {
    if (err) throw err;
    console.log('File Renamed!');
});
 

Append to Files Asynchronously

for (let i = 0; i < 10; i++) {
    fs.appendFile('testWrite.txt', i + '\n', function (err) {
        if (err) {
            console.log('We got an error  ' + err);
        }
    })
}
The above code will be executed asynchronously.  Look at the content of testWrite.txt
